AJAX

  • AJAX

AJAX是一种JS异步处理数据的技术，通过AJAX可以从服务器中读取相应信息而且是异步进行的，只有当数据接收并解析完毕之后才传回JS进行进一步处理，防止因同步发生信息阻塞，使用户无法进行下一步操作，下面是创建ajax对象的过程，通过调用这个函数，我们便可以以异步方式读取数据了。

function ajax(url,fnSucc,fnFain){
    //创建ajax对象
    var oAjax = null;
    if(window.XMLHttpRequest){
        oAjax =new XMLHttpRequest()
    }
    else{//兼容IE6
        oAjax = new ActiveXObject()
    }

    //和服务器连接，有get和post方法两种，ture表示异步，后面加Date是为了防止浏览器访问缓存
    oAjax.open('GET',url+"?t="+new Date().getDate(),true);

    //发送请求
    oAjax.send();

    oAjax.onreadystatechange = function(){
        if(oAjax.readyState ==4){     //当从服务器返回的信息解析完毕后
            if(oAjax.status == 200){   //当服务器返回状态码为200时执行成功函数
                fnSucc(oAjax.responseText) //responseText是解析完的文件内容
            }
            else{
                fnFain(oAjax.responseText)
            }

        }
    }
}

上面是get方法，如果是post方法则中间两步有变更
//url后面不加参数
oAjax.open(‘POST’,url);

下面对get和post方法进行封装

/**
 *
 * @param string url  
 * @param object options  
 *      //type: string post或者get，可以有一个默认值
        //data: 发送的数据，为一个键值对象或者为一个用&连接的赋值字符串
        //onsuccess: function 成功时的调用函数
        //onfail: function 失败时的调用函数
 */
function ajax(url, options) {
    //创建ajax对象
    var oAjax = null;
    if (window.XMLHttpRequest) {
        oAjax = new XMLHttpRequest()
    }
    else{
    oAjax = new ActiveXObject()}


    var type = options.type.toUpperCase() || "GET";
    var date = options.date || "";
    var parm = null
    var onsuccess = options.onsuccess || null
    var onfail = options.onfail || function () {
            alert(oAjax.status)
        }
    if (typeof date === "object") {
        parm = [];
        for (var k in date) {
            if (date.hasOwnProperty) {
                parm.push(k + "=" + date[k])
            }
        }
        parm = parm.join("&")
    }
    if (type == "GET") {
        oAjax.open('GET', url + "?" + parm);
        console.log(url + "?" + parm)
        oAjax.send();
    }
    else {
        oAjax.open(type, url);
        oAjax.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
        oAjax.send(parm);
        console.log(parm)
    }

    oAjax.onreadystatechange = function () {
        if (oAjax.readyState == 4) {     //当从服务器返回的信息解析完毕后
            if (oAjax.status == 200) {   //当服务器返回状态码为200时执行成功函数
                onsuccess(oAjax.responseText) //responseText是解析完的文件内容
            }
            else {
                onfail(oAjax.responseText)
            }

        }
    }
}
